\begin{problem}{2024}{G}{新课标II卷}{11}{导数}

(\hspace{1cm})
\begin{choices}
\choice{{当 $a>1$ 时，$f(x)$ 有 3 个零点}}
\choice{{当 $a<0$ 时，$x=0$ 是 $f(x)$ 的极大值点}}

\choice{{存在 $a$，使得点 $(1, f(1))$ 为曲线 $y=f(x)$ 的对称中心}}
\end{choices}
