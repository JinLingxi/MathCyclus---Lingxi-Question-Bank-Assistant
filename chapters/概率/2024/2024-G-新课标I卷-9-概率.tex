（若随机变量 $Z$ 服从正态分布$N(\mu, \sigma^2)$，则 $P(Z<\mu+\sigma)\approx 0.8413$）
(\hspace{1cm})
\begin{choices}
\choice{{$P(X>2)>0.2$}}
\choice{{$P(X>2)<0.5$}}
\choice{{$P(Y>2)>0.5$}}
\choice{{$P(Y>2)<0.8$}}
\end{choices}
\end{problem}
