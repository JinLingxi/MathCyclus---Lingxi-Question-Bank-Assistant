\begin{problem}{2013}{G}{重庆卷（理）}{10}{向量}
在平面上，$\overrightarrow{AB_1}\perp\overrightarrow{AB_2}$，$\left|\overrightarrow{OB_1}\right|=\left|\overrightarrow{OB_2}\right|=1$，$\overrightarrow{AP}=\overrightarrow{AB_1}+\overrightarrow{AB_2}$．若 $\left|\overrightarrow{OP}\right|<\dfrac{1}{2}$，则 $\left|\overrightarrow{OA}\right|$ 的取值范围是（　　）

\begin{choices}
\choice{{$\left(0,\dfrac{\sqrt{5}}{2}\right]$}}
\choice{{$\left(\dfrac{\sqrt{5}}{2},\dfrac{\sqrt{7}}{2}\right]$}}
\choice{{$\left(\dfrac{\sqrt{5}}{2},\sqrt{2}\right]$}}
\choice{{$\left(\dfrac{\sqrt{7}}{2},\sqrt{2}\right]$}}
\end{choices}
\end{problem}
