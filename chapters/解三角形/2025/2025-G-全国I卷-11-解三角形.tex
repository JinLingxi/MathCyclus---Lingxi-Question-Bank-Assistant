\begin{problem}{2025}{G}{全国I卷}{11}{解三角形}
已知 $\triangle ABC$ 的面积为 $\dfrac{1}{4}$，若 $\cos 2A + \cos 2B + 2\sin C = 2$，$\cos A \cos B \sin C = \dfrac{1}{4}$，则
(\hspace{1cm})
\begin{choices}
\choice{{$\sin C = \sin^2 A + \sin^2 B$}}
\choice{{$AB=\sqrt{2}$}}
\choice{{$\sin A + \sin B = \dfrac{ \sqrt{6} }{2}$ }}
\choice{{$AC^2+BC^2=3$}}
\end{choices}
\end{problem}
