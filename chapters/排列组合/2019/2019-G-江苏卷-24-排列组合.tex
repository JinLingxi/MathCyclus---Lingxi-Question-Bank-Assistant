\begin{problem}{2019}{G}{江苏卷}{24}{排列组合}
{G}{新课标I卷}{24}{排列组合}
设 $(1+x)^n = a_0 + a_1x + a_2x^2 + \cdots + a_nx^n$，$n\geq 4$，$n\in\mathbb{N}^*$．已知 $a_2^2 = 2a_4a_1$．

（1）求 $n$ 的值；

（2）设 $(1+\sqrt{3})^n = a + b\sqrt{3}$，其中 $a,b\in\mathbb{N}^*$，求 $a^2 - 3b^2$ 的值．
\end{problem}
