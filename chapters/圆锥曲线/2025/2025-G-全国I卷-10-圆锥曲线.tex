\begin{problem}{2025}{G}{全国I卷}{10}{圆锥曲线}

(\hspace{1cm})
\begin{choices}
\choice{{$|AD|=|AF|$}}
\choice{{$|AE|=|AB|$}}
\choice{{$|AB| \geqslant 6$}}
\choice{{$|AE|\cdot|BE| \geqslant 18$}}
\end{choices}
\end{problem}
