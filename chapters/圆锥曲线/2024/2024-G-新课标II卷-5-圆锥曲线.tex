\begin{problem}{2024}{G}{新课标II卷}{5}{圆锥曲线}
已知曲线 $C: x^2+y^2=16(y>0)$，从 $C$ 上任意一点 $P$ 向 $x$ 轴作垂线 $PP'$，$P'$ 是垂足，则线段 $PP'$ 的中点的轨迹方程为
(\hspace{1cm})
\begin{choices}[2]
\choice{{$\dfrac{x^2}{16}+\dfrac{y^2}{4}=1(y>0)$}}
\choice{{$\dfrac{x^2}{16}+\dfrac{y^2}{8}=1(y>0)$}}
\choice{{$\dfrac{y^2}{16}+\dfrac{x^2}{4}=1(y>0)$}}
\choice{{$\dfrac{y^2}{16}+\dfrac{x^2}{8}=1(y>0)$}}
\end{choices}
\end{problem}
