\begin{problem}{2024}{G}{新课标II卷}{10}{圆锥曲线}

(\hspace{1cm})
\begin{choices}
\choice{{$l$ 与 $\odot A$ 相切}}
\choice{{当 $P,A,B$ 共线时，$|PQ|=\sqrt{15}$}}
\choice{{当 $|PB|=2$ 时，$PA\perp AB$}}
\choice{{满足 $|PA|=|PB|$ 的点 $P$ 有且仅有 2 个}}
\end{choices}
\end{problem}
