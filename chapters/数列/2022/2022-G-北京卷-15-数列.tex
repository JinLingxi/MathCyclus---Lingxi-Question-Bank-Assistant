\begin{problem}{2022}{G}{北京卷}{15}{数列}

\circled{1} $\{a_n\}$ 的第 2 项小于 3；

\circled{2} $\{a_n\}$ 为等比数列；

\circled{3} $\{a_n\}$ 为递减数列；

\circled{4} $\{a_n\}$ 中存在小于 $\dfrac{1}{100}$ 的项.
