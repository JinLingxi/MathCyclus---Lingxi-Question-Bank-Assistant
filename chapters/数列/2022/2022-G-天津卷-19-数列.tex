\begin{problem}{2022}{G}{天津卷}{19}{数列}

设 $\{a_n\}$ 是等差数列，$\{b_n\}$ 是等比数列，且 $a_1=b_1=a_2-b_2=a_3-b_3=1$.

(1) 求 $\{a_n\}$ 与 $\{b_n\}$ 的通项公式;

(2) 设 $\{a_n\}$ 的前 $n$ 项和为 $S_n$，求证：$(S_{n+1}+a_{n+1})b_n=S_{n+1} b_{n+1}-S_n b_n$;

(3) 求 $\sum\limits_{k=1}^{2n} [a_{n+1}-(-1)^k a_k] b_k$.

\end{problem}
