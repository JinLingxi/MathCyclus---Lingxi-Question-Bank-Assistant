\begin{problem}{2022}{G}{上海卷}{16}{数列}
已知等比数列 $\{a_n\}$ 的前 $n$ 项和为 $S_n$, 前 $n$ 项积为 $T_n$, 则下列选项判断正确的是 (\hspace{1cm})
\begin{choices}
\choice{{若 $S_{2022}>S_{2021}$, 则数列 $\{a_n\}$ 是递增数列}}
\choice{{若 $T_{2022}>T_{2021}$, 则数列 $\{a_n\}$ 是递增数列}}
\choice{{若数列 $\{S_n\}$ 是递增数列, 则 $a_{2022} \ge a_{2021}$}}
\choice{{若数列 $\{T_n\}$ 是递增数列, 则 $a_{2022} \ge a_{2021}$}}
\end{choices}
\end{problem}
