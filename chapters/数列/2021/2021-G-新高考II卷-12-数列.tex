\begin{problem}{2021}{G}{新高考II卷}{12}{数列}
设正整数 $n=a_0 \cdot 2^0 + a_1 \cdot 2 + \cdots + a_{k-1} \cdot 2^{k-1} + a_k \cdot 2^k$，其中 $a_i \in \{0,1\}$，记 $\omega(n)=a_0+a_1+\cdots+a_k$. 则
(\hspace{1cm})
\begin{choices}
\choice{{$\omega(2n)=\omega(n)$}}
\choice{{$\omega(2n+3)=\omega(n)+1$}}
\choice{{$\omega(8n+5)=\omega(4n+3)$}}
\choice{{$\omega(2^n-1)=n$}}
\end{choices}
\end{problem}
