\begin{problem}{2021}{G}{浙江卷}{10}{数列}
已知数列 $\{a_n\}$ 满足 $a_1=1, a_{n+1}=\dfrac{a_n}{1+\sqrt[3]{a_n}}(n\in \mathbf{N}^*)$. 记数列 $\{a_n\}$ 的前 $n$ 项和为 $S_n$，则
(\hspace{1cm})
\begin{choices}
\choice{{$\dfrac{3}{2} < S_{100} < 3$}}
\choice{{$3 < S_{100} < 4$}}
\choice{{$4 < S_{100} < \dfrac{9}{2}$}}
\choice{{$\dfrac{9}{2} < S_{100} < 5$}}
\end{choices}
\end{problem}
