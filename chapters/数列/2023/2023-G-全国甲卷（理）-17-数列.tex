\begin{problem}{2023}{G}{全国甲卷}{17}{数列}

设 $S_n$ 为数列 $\{a_n\}$ 的前 $n$ 项和，已知 $a_2=1, 2S_n=na_{n+1}$.

(1) 求 $\{a_n\}$ 的通项公式;

(2) 求数列 $\left\{\dfrac{a_{n+1}}{2^n}\right\}$ 的前 $n$ 项和 $T_n$.

\end{problem}
