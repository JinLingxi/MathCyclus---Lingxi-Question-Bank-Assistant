\begin{problem}{2023}{G}{北京卷}{10}{数列}
已知数列 $\{a_n\}$ 满足 $a_{n+1}=\dfrac{1}{4}(a_n-6)^3+6 (n=1,2,3,\cdots)$，则
(\hspace{1cm})
\begin{choices}
\choice{{当 $a_1=3$ 时，$\{a_n\}$ 为递减数列，且存在常数 $M\leqslant 0$，使得 $a_n > M$ 恒成立}}
\choice{{当 $a_1=5$ 时，$\{a_n\}$ 为递增数列，且存在常数 $M\leqslant 6$，使得 $a_n < M$ 恒成立}}
\choice{{当 $a_1=7$ 时，$\{a_n\}$ 为递减数列，且存在常数 $M>6$，使得 $a_n > M$ 恒成立}}
\choice{{当 $a_1=9$ 时，$\{a_n\}$ 为递增数列，且存在常数 $M>0$，使得 $a_n < M$ 恒成立}}
\end{choices}
\end{problem}
