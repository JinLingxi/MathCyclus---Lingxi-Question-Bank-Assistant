\begin{problem}{2015}{G}{北京卷}{6}{数列}
设 $\{a_n\}$ 是等差数列．下列结论中正确的是
\begin{choices}
\choice{{若 $a_1+a_2>0$，则 $a_2+a_3>0$}}
\choice{{若 $a_1+a_3<0$，则 $a_1+a_2<0$}}
\choice{{若 $0<a_1<a_2$，则 $a_2>\sqrt{a_1a_3}$}}
\choice{{若 $a_1<0$，则 $(a_2-a_1)(a_2-a_3)>0$}}
\end{choices}
\end{problem}
