\begin{problem}{2025}{G}{全国II卷}{9}{数列}
记 $S_n$ 为等比数列 $\{a_n\}$ 的前 $n$ 项和，$q$ 为 $\{a_n\}$ 的公比，$q>0$，若 $S_3=7, a_3=1$，则
(\hspace{1cm})
\begin{choices}
\choice{{$q=\dfrac{1}{2}$}}
\choice{{$a_5=\dfrac{1}{9}$}}
\choice{{$S_5=8$}}
\choice{{$a_n+S_n=8$}}
\end{choices}
\end{problem}
