\begin{problem}{2018}{G}{浙江卷}{10}{数列}
已知 $a_1,a_2,a_3,a_4$ 成等比数列, 且 $a_1+a_2+a_3+a_4=\ln(a_1+a_2+a_3)$. 若 $a_1>1$, 则 (\hspace{1cm})
\begin{choices}
\choice{{$a_1<a_3,a_2<a_4$}}
\choice{{$a_1>a_3,a_2<a_4$}}
\choice{{$a_1<a_3,a_2>a_4$}}
\choice{{$a_1>a_3,a_2>a_4$}}
\end{choices}
\end{problem}
