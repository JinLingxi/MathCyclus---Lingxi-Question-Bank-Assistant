\begin{problem}{2025}{G}{全国II卷}{10}{函数}
已知 $f(x)$ 是定义在 $\mathbf{R}$ 上奇函数，当 $x>0$ 时，$f(x)=(x^2-3)e^x+2$，则
(\hspace{1cm})
\begin{choices}
\choice{{$f(0)=0$}}
\choice{{当 $x<0$ 时，$f(x)=(3-x^2)e^{-x}-2$}}
\choice{{$f(x) \geqslant 2$ 当且仅当 $x \geqslant \sqrt{3}$}}
\choice{{$x=-1$ 是 $f(x)$ 的极大值点}}
\end{choices}
\end{problem}
