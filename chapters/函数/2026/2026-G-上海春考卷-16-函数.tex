\begin{problem}{2026}{G}{上海春考卷}{16}{函数}
{G}{上海春季卷}{16}{函数}
对函数 $y=f(x),x\in\mathbb{R}$，定义集合
$$
A_f=\{(x,y)\mid y\ge f(x),x\in\mathbb{R}\}.
$$
对平面点集 $M$，若存在 $(x_0,y_0)\in M$，使对任意 $(x,y)\in M$，都有 $y\ge y_0$，则称 $(x_0,y_0)$ 为 $M$ 的“最低点”．对函数 $y=f(x)$ 与 $y=g(x)$，下列说法中正确的为（　　）

\begin{choices}
\choice{{若 $y=f(x)$ 和 $y=g(x)$ 都有最小值，则 $A_f\cap A_g$ 有最低点}}
\choice{{若 $A_f\cap A_g$ 有最小值，则 $y=f(x)$ 和 $y=g(x)$ 都有最小值}}
\choice{{若 $y=f(x)$ 或 $y=g(x)$ 有最小值，则 $A_f\cup A_g$ 有最低点}}
\choice{{若 $A_f\cup A_g$ 有最低点，则 $y=f(x)$ 或 $y=g(x)$ 有最小值}}
\end{choices}
\end{problem}
